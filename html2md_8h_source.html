49  if (amount == 0) return "";
50  else if (amount == 1) return str;
51 
52  std::string out;

281  struct TagAnchor : Tag {
282  void OnHasLeftOpeningTag(Converter* converter) override {
283  if (converter->IsInIgnoredTag()) return;
284 
285  if (converter->prev_tag_ == kTagImg) converter->AppendToMd('\n');
286 
287  converter->current_title_ = converter->ExtractAttributeFromTagLeftOf(kAttributeTitle);
288 
289  converter->current_href_ =
290  converter->RTrim(&converter->md_, true)
291  ->AppendBlank()
292  ->AppendToMd("[")
293  ->ExtractAttributeFromTagLeftOf(kAttributeHref);
294  }
295 
296  void OnHasLeftClosingTag(Converter* converter) override {
297  if (converter->IsInIgnoredTag()) return;
298 
299  if (converter->prev_ch_in_md_ == ' ') {
300  converter->ShortenMarkdown();
301  }
302 
303  if (converter->prev_ch_in_md_ == '[') {
304  converter->ShortenMarkdown();
305  } else {
306  converter->AppendToMd("](")
307  ->AppendToMd(converter->current_href_);
308 
309  // If title is set append it
310  if (!converter->current_title_.empty()) {
311  converter->AppendToMd(" \"")
312  ->AppendToMd(converter->current_title_)
313  ->AppendToMd('"');
314  }
315 
316  converter->AppendToMd(") ");
317 
318  if (converter->prev_tag_ == kTagImg)
319  converter->AppendToMd('\n');
320  }
321  }

569 // Here we can improve the performance
570 [[nodiscard]] bool IsInIgnoredTag() const {
571 auto len = dom_tags_.size();
572
573 for (auto i = 0; i < len; ++i) {
574 std::string tag = dom_tags_[i];
575
576 if (tag == kTagTitle && !option.includeTitle)
577 return true;
578
579 if (tag == kTagPre || tag == kTagTitle)
580 return false;
581
582 if (IsIgnoredTag(tag))
583 return true;
584 }
585
586 return false;
